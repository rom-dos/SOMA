upper = \relative c' {
  \clef treble
  \key c \major
  \time 4/4

      <a f g d> |
  }

  lower = \relative c {
  \clef bass
  \key c \major
  \time 4/4

    r1 |  
  }

  \score {
    \new PianoStaff \with { instrumentName = #"Piano" }
    <<
      \new Staff = "upper" \upper
      \new Staff = "lower" \lower
    >>
    \layout { }
    \midi { }
  }
